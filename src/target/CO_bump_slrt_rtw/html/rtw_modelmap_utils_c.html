45/** Function: rtwCAPI_EncodePath ===============================================



46 *  Abstract:



47 *     Escape all '|' characters in bpath. For examples 'aaa|b' will become



48 *     'aaa~|b'. The caller is responsible for freeing the returned string



49 *



50 *



51 * NOTE: returned string can be NULL in two cases:



52 *     (1) string passed in was NULL



53 *     (2) a memory allocation error occurred



54 * In the second case, the caller need to report the error



55 */



56char* rtwCAPI_EncodePath(const char* path)



57{



58    char* encodedPath     = NULL;



59    size_t pathLen        = (path==NULL) ? 0:strlen(path) + 1;



60    size_t encodedPathLen = pathLen;



61    unsigned i;



62    unsigned j = 0;



63



64    if (path == NULL) return NULL;



65



66    for (i = 0; i < pathLen; ++i) {



67        if (path[i] == '|' || path[i] == '~') ++encodedPathLen;



68    }



69



70    encodedPath = (char_T*)utMalloc(encodedPathLen*sizeof(char_T));



71    if (encodedPath == NULL) return encodedPath;



72



73    for (i = 0; i < pathLen; ++i) {



74        char ch = path[i];



75        if (ch == '~' || ch == '|') encodedPath[j++] = '~';



76        encodedPath[j++] = ch;



77    }



78    utAssert(j == encodedPathLen);



79    utAssert(encodedPath[j-1] == '\0');



80



81    return encodedPath;



82



83} /* rtwCAPI_EncodePath */

85/** Function: rtwCAPI_GetSigAddrFromMap ========================================



86 *



87 */



88void rtwCAPI_GetSigAddrFromMap(uint8_T      isPointer,



89                               int_T*       sigComplexity,



90                               int_T*       sigDataType,



91                               void**       sigDataAddr,



92                               int_T*       sigIdx,



93                               uint_T       mapIdx,



94                               void**       dataAddrMap)



95{



96    if (isPointer) {



97        /* Dereference pointer and cache the address */



98



99        /* Imported Pointers cannot be complex - Assert */



100        utAssert(sigComplexity[*sigIdx] != 1);



101        UNUSED_PARAMETER(sigComplexity);



102



103        /* Check for data type and dereference accordingly */



104        switch (sigDataType[*sigIdx]) {



105          case SS_DOUBLE:



106            sigDataAddr[*sigIdx] = \



107                (void*) *((real_T **) dataAddrMap[mapIdx]);



108            break;



109          case SS_SINGLE:



110            sigDataAddr[*sigIdx] = \



111                (void*) *((real32_T **) dataAddrMap[mapIdx]);



112            break;



113          case SS_UINT32:



114            sigDataAddr[*sigIdx] = \



115                (void*) *((uint32_T **) dataAddrMap[mapIdx]);



116            break;



117          case SS_INT32:



118            sigDataAddr[*sigIdx] = \



119                (void*) *((int32_T **) dataAddrMap[mapIdx]);



120            break;



121          case SS_UINT16:



122            sigDataAddr[*sigIdx] = \



123                (void*) *((uint16_T **) dataAddrMap[mapIdx]);



124            break;



125          case SS_INT16:



126            sigDataAddr[*sigIdx] = \



127                (void*) *((int16_T **) dataAddrMap[mapIdx]);



128            break;



129          case SS_UINT8:



130            sigDataAddr[*sigIdx] = \



131                (void*) *((uint8_T **) dataAddrMap[mapIdx]);



132            break;



133          case SS_INT8:



134            sigDataAddr[*sigIdx] = \



135                (void*) *((int8_T **) dataAddrMap[mapIdx]);



136            break;



137          case SS_BOOLEAN:



138            sigDataAddr[*sigIdx] = \



139                (void*) *((boolean_T **) dataAddrMap[mapIdx]);



140            break;



141          default:



142            sigDataAddr[*sigIdx] = \



143                (void*) *((real_T **) dataAddrMap[mapIdx]);



144            break;



145        }  /* end switch */



146    } else {



147        /* if Data is not a pointer store the address directly */



148        sigDataAddr[*sigIdx] = dataAddrMap[mapIdx];



149    }



150



151} /* rtwCAPI_GetSigAddrFromMap */

154/** Function: rtwCAPI_HasStates ================================================

157boolean_T rtwCAPI_HasStates(const rtwCAPI_ModelMappingInfo* mmi)



158{



159    int_T i;



160    int_T nCMMI;



161



162    if (mmi == NULL) return(0U);



163



164    if (rtwCAPI_GetNumStates(mmi) > 0) return(1U);



165



166    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



167    for (i = 0; i < nCMMI; ++i) {



168        if (rtwCAPI_HasStates(rtwCAPI_GetChildMMI(mmi,i))) return(1U);



169    }



170    return(0U);



171



172} /* rtwCAPI_HasStates */

239    int_T i;

241    int_T nCMMI;

244    const rtwCAPI_DataTypeMap* dataTypeMap;

263    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);

265        const rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



266        nCStateRecs += rtwCAPI_GetNumContStateRecords(cMMI);



267    }



268    return nCStateRecs;

298/** Function: rtwCAPI_UpdateFullPaths =========================================*



299 *



300 */



301const char_T* rtwCAPI_UpdateFullPaths(rtwCAPI_ModelMappingInfo* mmi,



302                                      const char_T* path,



303                                      boolean_T isCalledFromTopModel)



304{



305    int_T         i;



306    int_T         nCMMI;



307    size_t        pathLen;



308    char_T*       mmiPath;



309    size_t        mmiPathLen;



310    char_T*       relMMIPath;



311    size_t        relMMIPathLen;



312



313    if (mmi == NULL) return NULL;



314



315    utAssert(path != NULL);



316    utAssert( rtwCAPI_GetFullPath(mmi) == NULL );



317



318    pathLen = strlen(path)+1;



319



320    if (isCalledFromTopModel) {



321        /* If called from top model - FullPath is same as path */



322        mmiPath = (char_T*)utMalloc(pathLen*sizeof(char_T));



323        (void)memcpy(mmiPath, path, pathLen*sizeof(char_T));



324    }



325    else {        



326        relMMIPath = rtwCAPI_EncodePath(rtwCAPI_GetPath(mmi));



327        if ( (relMMIPath== NULL) && (rtwCAPI_GetPath(mmi) != NULL)) {



328            return rtwCAPI_mallocError;



329        }



330        relMMIPathLen = relMMIPath ? (strlen(relMMIPath) + 1) : 0;



331        



332        mmiPathLen = pathLen + relMMIPathLen;



333        



334        mmiPath = (char_T*)utMalloc(mmiPathLen*sizeof(char_T));



335        if (mmiPath == NULL) return rtwCAPI_mallocError;



336        (void)memcpy(mmiPath, path, pathLen*sizeof(char_T));



337        utAssert(mmiPath[pathLen-1] == '\0');



338        



339        if (relMMIPath) {



340            /* mmiPath = path + | + relMMIPath + '\0' */



341            mmiPath[pathLen-1] = '|';



342            (void)memcpy(&(mmiPath[pathLen]),



343                         relMMIPath, relMMIPathLen*sizeof(char_T));



344            utAssert(mmiPath[mmiPathLen-1] == '\0');



345            utFree(relMMIPath);



346        }



347    }



348    rtwCAPI_SetFullPath(*mmi, mmiPath);



349



350    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



351    for (i = 0; i < nCMMI; ++i) {



352        rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



353        const char_T* errstr = rtwCAPI_UpdateFullPaths(cMMI, mmiPath, 0);



354        if (errstr != NULL) return errstr;



355    }



356    return NULL;



357



358} /* rtwCAPI_UpdateFullPaths */

361/** Function: rtwCAPI_GetFullStateBlockPath ===================================



362 *



363 */



364char* rtwCAPI_GetFullStateBlockPath(const char* stateBlockPath,



365                                           const char* mmiPath,



366                                           size_t      mmiPathLen,



367                                           boolean_T   crossingModel)



368{



369    char_T* blockPath          = NULL;



370    char_T* fullStateBlockPath = NULL;



371    size_t     fullStateBlockPathLen;



372



373    if (stateBlockPath == NULL) goto EXIT_POINT;



374



375    /* fullStateBlockPath = mmiPath + | + blockPath + '\0' */



376    /* If crossing a model boundary encode, otherwise do not */



377



378    if (crossingModel) {



379        blockPath = rtwCAPI_EncodePath(stateBlockPath);



380        if (blockPath == NULL) goto EXIT_POINT;



381    } else {



382        size_t len = strlen(stateBlockPath)+1;



383        blockPath = (char*)utMalloc(len*sizeof(char));



384        if (blockPath == NULL) goto EXIT_POINT;



385        (void)strcpy(blockPath,stateBlockPath);



386    }



387    utAssert(blockPath != NULL);



388    fullStateBlockPathLen = ( (mmiPath==NULL) ?



389                              strlen(blockPath) + 1 :



390                              mmiPathLen + strlen(blockPath) + 2 );



391    fullStateBlockPath    = (char*)utMalloc(fullStateBlockPathLen*sizeof(char));



392    if (fullStateBlockPath == NULL) goto EXIT_POINT;



393



394    if (mmiPath != NULL) {



395        (void)strcpy(fullStateBlockPath, mmiPath);



396        fullStateBlockPath[mmiPathLen]   = '|';



397        fullStateBlockPath[mmiPathLen+1] =  '\0';



398        (void)strcat(fullStateBlockPath, blockPath);



399    } else {



400        (void)strcpy(fullStateBlockPath, blockPath);



401        fullStateBlockPath[fullStateBlockPathLen-1] =  '\0';



402    }



403    utAssert(fullStateBlockPath[fullStateBlockPathLen-1] == '\0');



404



405  EXIT_POINT:



406    utFree(blockPath);



407    return fullStateBlockPath; /* caller is responsible for free */



408}

410uint_T rtwCAPI_GetStateWidth(const rtwCAPI_DimensionMap* dimMap,



411                             const uint_T*               dimArray,



412                             const rtwCAPI_States*       states,



413                             uint_T                      iState)



414{



415    uint_T mapIdx = rtwCAPI_GetStateDimensionIdx(states, iState);



416    utAssert( rtwCAPI_GetNumDims(dimMap,mapIdx) == 2 );



417    mapIdx = rtwCAPI_GetDimArrayIndex(dimMap, mapIdx);



418    return (dimArray[mapIdx] * dimArray[mapIdx+1]);



419}

423/** Function: rtwCAPI_GetStateRecordInfo =======================================



424 *



425 */



426const char_T* rtwCAPI_GetStateRecordInfo(const rtwCAPI_ModelMappingInfo* mmi,



427                                         const char_T**    sigBlockName,



428                                         const char_T**    sigLabel,



429                                         const char_T**    sigName,



430                                         int_T*            sigWidth,



431                                         int_T*            sigDataType,



432                                         int_T*            logDataType,



433                                         int_T*            sigComplexity,



434                                         void**            sigDataAddr,



435                                         boolean_T*        sigCrossMdlRef,



436                                         boolean_T*        sigInProtectedMdl,



437                                         const char_T**    sigPathAlias,



438                                         real_T*           sigSampleTime,



439                                         int_T*            sigHierInfoIdx,



440                                         uint_T*           sigFlatElemIdx,                                         



441                                         const rtwCAPI_ModelMappingInfo** sigMMI,



442                                         int_T*            sigIdx,



443                                         boolean_T         crossingModel,



444                                         boolean_T         isInProtectedMdl,



445                                         real_T*           contStateDeriv,



446                                         boolean_T         rtwLogging)



447{



448    int_T               i;



449    int_T               nCMMI;



450    int_T               nStates;



451    const char_T*       mmiPath;



452    size_t              mmiPathLen;



453    const rtwCAPI_States*  states;



454    const rtwCAPI_DimensionMap* dimMap;



455    const uint_T*       dimArray;



456    const rtwCAPI_DataTypeMap*  dataTypeMap;



457    void**              dataAddrMap;



458    const char_T*       errstr = NULL;



459    uint8_T             isPointer = 0;



460



461    if (mmi == NULL) goto EXIT_POINT;



462    isInProtectedMdl = isInProtectedMdl || rtwCAPI_IsProtectedModel(mmi);



463    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



464    for (i = 0; i < nCMMI; ++i) {



465        rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



466        real_T* childContStateDeriv = NULL;



467



468        if (cMMI == NULL) continue;



469



470        if (contStateDeriv) {



471            int idx;



472



473            idx = rtwCAPI_MMIGetContStateStartIndex(cMMI);



474            if(idx == -1) continue;



475            



476            childContStateDeriv = &contStateDeriv[idx];



477        }



478        errstr = rtwCAPI_GetStateRecordInfo(cMMI,



479                                            sigBlockName,



480                                            sigLabel,



481                                            sigName,



482                                            sigWidth,



483                                            sigDataType,



484                                            logDataType,



485                                            sigComplexity,



486                                            sigDataAddr,



487                                            sigCrossMdlRef,



488                                            sigInProtectedMdl,



489                                            sigPathAlias,



490                                            sigSampleTime,



491                                            sigHierInfoIdx,



492                                            sigFlatElemIdx,



493                                            sigMMI,



494                                            sigIdx,



495                                            0x1, /* true, */



496                                            isInProtectedMdl,



497                                            childContStateDeriv,



498                                            rtwLogging);



499        if (errstr != NULL) goto EXIT_POINT;



500    }



501



502    nStates = rtwCAPI_GetNumStates(mmi);



503    if (nStates < 1) goto EXIT_POINT;



504



505    mmiPath     = rtwCAPI_GetFullPath(mmi);



506    mmiPathLen  = (mmiPath==NULL)? 0 : strlen(mmiPath);



507    states      = rtwCAPI_GetStates(mmi);



508    dimMap      = rtwCAPI_GetDimensionMap(mmi);



509    dimArray    = rtwCAPI_GetDimensionArray(mmi);



510    dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);



511    dataAddrMap = rtwCAPI_GetDataAddressMap(mmi);



512



513    for (i = 0; i < nStates; ++i) {



514        uint_T      mapIdx;



515



516        /* If collecting continuous states, skip non-continuous states */



517        if (contStateDeriv && !rtwCAPI_IsAContinuousState(states,i)) continue;



518



519        /* For RTW logging, skip states that cannot be logged to MAT-File. */



520        if ((rtwLogging) &&



521            (rtwCAPI_CanLogStateToMATFile(dataTypeMap, states, i) == false)) continue;



522



523        /* BlockPath (caller is responsible for free) */



524        sigBlockName[*sigIdx] =



525            rtwCAPI_GetFullStateBlockPath(rtwCAPI_GetStateBlockPath(states,i),



526                                          mmiPath, mmiPathLen, crossingModel);



527        if (sigBlockName[*sigIdx] == NULL) {



528            errstr = rtwCAPI_mallocError;



529            goto EXIT_POINT;



530        }



531



532        /* Label */



533        if (rtwCAPI_IsAContinuousState(states,i)){



534            sigLabel[*sigIdx] = "CSTATE";



535        } else {



536            sigLabel[*sigIdx] = "DSTATE";



537        }



538        sigName[*sigIdx] = rtwCAPI_GetStateName(states, i);



539



540        /* Width */



541        sigWidth[*sigIdx] = rtwCAPI_GetStateWidth(dimMap, dimArray, states, i);



542



543        /* DataType and logDataType */



544        mapIdx = rtwCAPI_GetStateDataTypeIdx(states, i);



545        sigDataType[*sigIdx] = rtwCAPI_GetDataTypeSLId(dataTypeMap, mapIdx);



546        /* this mimics code in simulink.dll:DtGetDataTypeLoggingId() */



547        if (logDataType) {



548            switch (sigDataType[*sigIdx]) {



549              case SS_DOUBLE:



550              case SS_SINGLE:



551              case SS_INT8:



552              case SS_UINT8:



553              case SS_INT16:



554              case SS_UINT16:



555              case SS_INT32:



556              case SS_UINT32:



557              case SS_BOOLEAN:



558                logDataType[*sigIdx] = sigDataType[*sigIdx];



559                break;



560              case SS_ENUM_TYPE:



561                logDataType[*sigIdx] = SS_INT32;



562                break;



563              default:



564                logDataType[*sigIdx] = SS_DOUBLE;



565                break;



566            }



567        }



568



569        /* Complexity */



570        sigComplexity[*sigIdx] = rtwCAPI_GetDataIsComplex(dataTypeMap, mapIdx);



571



572        /* Data Access - Pointer or Direct*/



573        isPointer = ((uint8_T)rtwCAPI_GetDataIsPointer(dataTypeMap, mapIdx));



574



575        /* Address */



576        if (contStateDeriv) {



577            int_T csvIdx = rtwCAPI_GetContStateStartIndex(states,i);



578            utAssert(csvIdx >= 0);



579            sigDataAddr[*sigIdx] = &contStateDeriv[csvIdx];



580        } else {



581            mapIdx = rtwCAPI_GetStateAddrIdx(states,i);



582            rtwCAPI_GetSigAddrFromMap(isPointer, sigComplexity, sigDataType,



583                                      sigDataAddr, sigIdx, mapIdx, dataAddrMap);



584        }



585



586        /* CrossingModelBoundary */



587        sigCrossMdlRef[*sigIdx] = crossingModel;



588



589        if (sigInProtectedMdl)



590        {



591            sigInProtectedMdl[*sigIdx] = isInProtectedMdl;



592        }



593



594        if (sigPathAlias && 



595            rtwCAPI_GetStatePathAlias(states,i) != NULL && 



596            rtwCAPI_GetStatePathAlias(states,i)[0] != '\0') {



597            sigPathAlias[*sigIdx] =  



598                rtwCAPI_GetFullStateBlockPath(rtwCAPI_GetStatePathAlias(states,i),



599                                              mmiPath, mmiPathLen, crossingModel);



600        }



601        



602        /* Sample Time */



603        if (sigSampleTime) {



604            const rtwCAPI_SampleTimeMap* tsMap = rtwCAPI_GetSampleTimeMap(mmi);



605            int_T TID;



606            mapIdx = rtwCAPI_GetStateSampleTimeIdx(states, i);



607            TID = rtwCAPI_GetSampleTimeTID(tsMap, mapIdx);



608            if (TID >= 0) {



609                sigSampleTime[2*(*sigIdx)] =



610                    *((const real_T*)rtwCAPI_GetSamplePeriodPtr(tsMap,mapIdx));



611                sigSampleTime[2*(*sigIdx)+1] =



612                    *((const real_T*)rtwCAPI_GetSampleOffsetPtr(tsMap,mapIdx));



613            } else { /* triggered */



614                utAssert(TID==-1); 



615                sigSampleTime[2*(*sigIdx)]   = -1.0;



616                sigSampleTime[2*(*sigIdx)+1] = -1.0;



617            }



618        }



619



620        /* HierInfoIdx and FlatElemIdx */



621        if (sigHierInfoIdx && sigFlatElemIdx)



622        {



623            sigHierInfoIdx[*sigIdx] = rtwCAPI_GetStateHierInfoIdx(states, i);



624            sigFlatElemIdx[*sigIdx] = rtwCAPI_GetStateFlatElemIdx(states, i);



625        }      



626



627        /* MMI for each state */



628        if (sigMMI)



629        {



630            sigMMI[*sigIdx] = mmi;



631        }        



632



633        ++(*sigIdx);



634    }



635



636  EXIT_POINT:



637    return(errstr);



638



639} /* rtwCAPI_GetStateRecordInfo */

643/** Function: rtwCAPI_GetNumSigLogRecords ======================================



644 *



645 */



646int_T rtwCAPI_GetNumSigLogRecords(const rtwCAPI_ModelMappingInfo* mmi)



647{

649    int_T nRecs;



650    int_T nCMMI;



651



652    if (mmi == NULL) return(0);



653

701/** Function: rtwCAPI_GetSigLogRecordInfo ======================================



702 *



703 */



704const char_T* rtwCAPI_GetSigLogRecordInfo(const rtwCAPI_ModelMappingInfo* mmi,



705                                          const char_T**    sigBlockName,



706                                          const char_T**    sigLabel,



707                                          int_T*            sigWidth,



708                                          int_T*            sigDataType,



709                                          int_T*            logDataType,



710                                          int_T*            sigComplexity,



711                                          void**            sigDataAddr,



712                                          boolean_T*        sigCrossMdlRef,



713                                          int_T*            sigIdx,



714                                          boolean_T         crossingModel,



715                                          boolean_T         rtwLogging)



716{



717    int_T               i;



718    int_T               nCMMI;



719    int_T               nSignals;



720    const char_T*       mmiPath;



721    size_t              mmiPathLen;



722    const rtwCAPI_Signals*  signals;



723    const rtwCAPI_DimensionMap* dimMap;



724    const uint_T*       dimArray;



725    const rtwCAPI_DataTypeMap*  dataTypeMap;



726    void**              dataAddrMap;



727    const char_T*       errstr = NULL;



728    uint8_T             isPointer = 0;



729    char*               blockPath = NULL;



730



731    if (mmi == NULL) goto EXIT_POINT;



732



733    nCMMI = rtwCAPI_GetChildMMIArrayLen(mmi);



734    for (i = 0; i < nCMMI; ++i) {



735        rtwCAPI_ModelMappingInfo* cMMI = rtwCAPI_GetChildMMI(mmi,i);



736



737        errstr = rtwCAPI_GetSigLogRecordInfo(cMMI,



738                                             sigBlockName,



739                                             sigLabel,



740                                             sigWidth,



741                                             sigDataType,



742                                             logDataType,



743                                             sigComplexity,



744                                             sigDataAddr,



745                                             sigCrossMdlRef,



746                                             sigIdx,



747                                             true,



748                                             rtwLogging);



749        if (errstr != NULL) goto EXIT_POINT;



750    }



751



752    nSignals = rtwCAPI_GetNumSignals(mmi);



753    if (nSignals < 1) goto EXIT_POINT;



754



755    mmiPath     = rtwCAPI_GetFullPath(mmi);



756    mmiPathLen  = (mmiPath==NULL)? 0 : strlen(mmiPath);



757    signals     = rtwCAPI_GetSignals(mmi);



758    dimMap      = rtwCAPI_GetDimensionMap(mmi);



759    dimArray    = rtwCAPI_GetDimensionArray(mmi);



760    dataTypeMap = rtwCAPI_GetDataTypeMap(mmi);



761    dataAddrMap = rtwCAPI_GetDataAddressMap(mmi);



762



763    for (i = 0; i < nSignals; ++i) {



764        uint_T mapIdx;



765        size_t sigPathLen;



766        char*  sigPath;



767



768        /* For RTW logging, skip states that cannot be logged to MAT-File. */



769        if ((rtwLogging) &&



770            (rtwCAPI_CanLogSignalToMATFile(dataTypeMap, signals, i) == false)) continue;



771



772        /* sigBlockPath = mmiPath + | + BlockPath + '\0' */



773        /* If crossing a model boundary encode, otherwise do not */



774



775        if (crossingModel) {



776            blockPath = rtwCAPI_EncodePath(rtwCAPI_GetSignalBlockPath(signals, i));



777            if ( (blockPath == NULL) &&



778                 (rtwCAPI_GetSignalBlockPath(signals, i) != NULL)) {



779                errstr = rtwCAPI_mallocError;



780                goto EXIT_POINT;



781            }



782        } else {



783            const char* constBlockPath = rtwCAPI_GetSignalBlockPath(signals, i);



784            blockPath = (char*)utMalloc((strlen(constBlockPath)+1)*sizeof(char));



785            (void)strcpy(blockPath, constBlockPath);



786        }



787        utAssert(blockPath != NULL);



788        sigPathLen = ( (mmiPath==NULL) ?



789                                   strlen(blockPath) + 1 :



790                                   mmiPathLen + strlen(blockPath) + 2 );



791        sigPath    = (char*)utMalloc(sigPathLen*sizeof(char));



792        if (sigPath == NULL) {



793            errstr = rtwCAPI_mallocError;



794            goto EXIT_POINT;



795        }



796        if (mmiPath != NULL) {



797            (void)strcpy(sigPath, mmiPath);



798            sigPath[mmiPathLen]   = '|';



799            sigPath[mmiPathLen+1] =  '\0';



800            (void)strcat(sigPath, blockPath);



801        } else {



802            (void)strcpy(sigPath, blockPath);



803            sigPath[sigPathLen-1] =  '\0';



804        }



805       /* need to free for every iteration of the loop, but also have



806        * the free below EXIT_POINT in case of error */



807        utFree(blockPath);



808        blockPath = NULL;



809        utAssert(sigPath[sigPathLen-1] == '\0');



810        sigBlockName[*sigIdx] = sigPath; /* caller is responsible for free */



811



812        /* Label */



813        sigLabel[*sigIdx] = rtwCAPI_GetSignalName(signals, i);



814



815        /* Width */



816        mapIdx = rtwCAPI_GetSignalDimensionIdx(signals, i);



817        utAssert( rtwCAPI_GetNumDims(dimMap,mapIdx) == 2 );



818        mapIdx = rtwCAPI_GetDimArrayIndex(dimMap, mapIdx);



819        sigWidth[*sigIdx] = dimArray[mapIdx] * dimArray[mapIdx+1];



820



821        /* DataType and logDataType */



822        mapIdx = rtwCAPI_GetSignalDataTypeIdx(signals, i);



823        sigDataType[*sigIdx] = rtwCAPI_GetDataTypeSLId(dataTypeMap, mapIdx);



824        /* this mimics code in simulink.dll:mapSigDataTypeToLogDataType */



825        switch (sigDataType[*sigIdx]) {



826          case SS_DOUBLE:



827          case SS_SINGLE:



828          case SS_INT8:



829          case SS_UINT8:



830          case SS_INT16:



831          case SS_UINT16:



832          case SS_INT32:



833          case SS_UINT32:



834          case SS_BOOLEAN:



835            logDataType[*sigIdx] = sigDataType[*sigIdx];



836            break;



837          case SS_ENUM_TYPE:



838            logDataType[*sigIdx] = SS_INT32;



839            break;



840          default:



841            logDataType[*sigIdx] = SS_DOUBLE;



842            break;



843        }



844



845        /* Complexity */



846        sigComplexity[*sigIdx] = rtwCAPI_GetDataIsComplex(dataTypeMap, mapIdx);



847



848        /* Data Access - Pointer or Direct*/



849        isPointer = ((uint8_T)rtwCAPI_GetDataIsPointer(dataTypeMap, mapIdx));



850



851        /* Address */



852        mapIdx = rtwCAPI_GetSignalAddrIdx(signals, i);



853



854        rtwCAPI_GetSigAddrFromMap(isPointer, sigComplexity, sigDataType,



855                                  sigDataAddr, sigIdx, mapIdx, dataAddrMap);



856



857        /* CrossingModelBoundary */



858        sigCrossMdlRef[*sigIdx] = crossingModel;



859



860        ++(*sigIdx);



861    }



862



863  EXIT_POINT:



864    utFree(blockPath);



865    return(errstr);



866



867} /* rtwCAPI_GetSigLogRecordInfo */

870/** Function: rtwCAPI_CountSysRan ==============================================



871 *   Recursive function that counts the number of non-NULL pointers in the array



872 *   of system ran dwork pointers, for the given MMI and below



873 */



874void rtwCAPI_CountSysRan(const rtwCAPI_ModelMappingInfo *mmi,



875			 int                            *count)



876{



877    sysRanDType **sysRan;



878    int numSys;



879    int nCMMI;



880    int numSysRan = 0;



881    int i;



882



883    if (mmi == NULL) return;



884



885    sysRan = rtwCAPI_GetSystemRan(mmi);



886    numSys    = rtwCAPI_GetNumSystems(mmi);



887    nCMMI     = rtwCAPI_GetChildMMIArrayLen(mmi);



888



889    /* Recurse over children */



890    for (i = 0; i < nCMMI; i++) {



891        rtwCAPI_CountSysRan(rtwCAPI_GetChildMMI(mmi,i), &numSysRan);



892    }



893



894    /* Count number of dworks in this MMI - skip root */



895    for (i = 1; i< numSys; i++) {



896        if (sysRan[i] != NULL) numSysRan++;



897    }



898



899    *count += numSysRan;



900



901} /* end rtwCAPI_CountSysRan */

904/** Function: rtwCAPI_FillSysRan ===============================================



905 *   Recursive function that fills in the system ran dwork pointers and their



906 *   corresponding tids in the given array, for the given MMI and below.  The



907 *   array to be filled in must be allocated outside



908 */



909void rtwCAPI_FillSysRan(const rtwCAPI_ModelMappingInfo *mmi,



910			sysRanDType                    **sysRan,



911			int                            *sysTid,



912                        int                            *fillIdx)



913{



914    int       numSys;



915    sysRanDType **mmiSysRan;



916    int       nCMMI;



917    int       *mmiSysTid;



918    int       idx         = *fillIdx;



919    const int *mmiConSys;



920    int i;



921



922    if (mmi == NULL) return;



923



924    numSys      = rtwCAPI_GetNumSystems(mmi);



925    mmiSysRan = rtwCAPI_GetSystemRan(mmi);



926    nCMMI       = rtwCAPI_GetChildMMIArrayLen(mmi);



927    mmiSysTid  = rtwCAPI_GetSystemTid(mmi);



928    mmiConSys  = rtwCAPI_GetContextSystems(mmi);



929



930    /* Recurse over children */



931    for (i = 0; i < nCMMI; i++) {



932        rtwCAPI_FillSysRan(rtwCAPI_GetChildMMI(mmi,i), sysRan, sysTid, &idx);



933    }



934



935    /* Populate arrays with dwork pointers and tid - skip root */



936    for (i = 1; i< numSys; i++) {



937        if (mmiSysRan[i] != NULL) {



938            idx++;



939            sysRan[idx] = mmiSysRan[i];



940            sysTid[idx] = mmiSysTid[mmiConSys[i]];



941        }



942    }



943



944    *fillIdx = idx;



945



946} /* end rtwCAPI_FillSysRan */
