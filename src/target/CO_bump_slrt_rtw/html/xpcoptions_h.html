29/* Change all stepsize using the newBaseRateStepSize */



30void CO_bump_ChangeStepSize(real_T newBaseRateStepSize, RT_MODEL_CO_bump_T *



31  const CO_bump_M)



32{



33  real_T ratio = newBaseRateStepSize / 0.001;



34



35  /* update non-zore stepsize of periodic



36   * sample time. Stepsize of asynchronous



37   * sample time is not changed in this function */



38  CO_bump_M->Timing.stepSize0 = CO_bump_M->Timing.stepSize0 * ratio;



39  CO_bump_M->Timing.stepSize1 = CO_bump_M->Timing.stepSize1 * ratio;



40  CO_bump_M->Timing.stepSize2 = CO_bump_M->Timing.stepSize2 * ratio;



41  CO_bump_M->Timing.stepSize = CO_bump_M->Timing.stepSize * ratio;



42}
